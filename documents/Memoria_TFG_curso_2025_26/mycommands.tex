  }%

\newcommand*{\@subsetsim}[2]{%
  \sbox0{$\m@th#1\subset$}%
  \sbox2{$\m@th#2\sim$}%
  \sbox4{$\m@th#2$}%
  \raisebox{.5\dimexpr\ht2-\ht4\relax}{%
    \oalign{\copy0 \cr\hidewidth\copy2\hidewidth}%
  }%
}
